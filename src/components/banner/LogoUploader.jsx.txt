import React, { useRef } from 'react';
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";
import { Upload, X } from "lucide-react";

export default function LogoUploader({ logo, onLogoChange, whatsapp, onWhatsappChange }) {
  const inputRef = useRef(null);

  const handleFile = (e) => {
    const file = e.target.files?.[0];
    if (!file) return;
    const reader = new FileReader();
    reader.onload = (ev) => onLogoChange(ev.target.result);
    reader.readAsDataURL(file);
  };

  return (
    <div className="flex items-center gap-4">
      <input
        ref={inputRef}
        type="file"
        accept="image/*"
        onChange={handleFile}
        className="hidden"
      />
      {logo ? (
        <div className="flex items-center gap-3 bg-white/5 rounded-lg px-3 py-2 ring-1 ring-white/10">
          <img src={logo} alt="Logo" className="h-8 w-auto object-contain" />
          <span className="text-white/60 text-sm">Logo carregada</span>
          <button onClick={() => onLogoChange(null)} className="text-white/40 hover:text-red-400 transition-colors">
            <X className="w-4 h-4" />
          </button>
        </div>
      ) : (
        <Button
          variant="outline"
          onClick={() => inputRef.current?.click()}
          className="border-white/10 bg-white/5 text-white/70 hover:bg-white/10 hover:text-white"
        >
          <Upload className="w-4 h-4 mr-2" />
          Carregar Logo
        </Button>
      )}
      
      <Input
        placeholder="WhatsApp (ex: 11999999999)"
        value={whatsapp}
        onChange={(e) => onWhatsappChange(e.target.value)}
        className="bg-white/5 border-white/10 text-white placeholder:text-white/40 w-64"
      />
    </div>
  );
}
